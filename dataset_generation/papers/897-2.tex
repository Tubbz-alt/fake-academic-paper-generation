\caption{Our Algorithm}\label{test_fly}
\begin{algorithmic}[1]
\While{No Crash}
    \State \textbf{Input : } Real time image
    \State{leftPred, straightPred, rightPred} = Network\{image\}
    \If{StraightPred $>$ $\alpha$ }
        \State Linear Velocity = $\beta$
        \State Angular Velocity $\propto$ Right Prediction - Left Prediction
    \Else
        \State Linear Velocity = 0
        \If{Right Prediction $>$ Left Prediction}
            \While{Straight Prediction $<$ $\alpha$}
                \State Take Right turn
        \EndWhile
        \Else
            \While{Straight Prediction $<$ $\alpha$}
            \State Take Left turn
        \EndWhile
        \EndIf 
    \EndIf
\EndWhile

\end{algorithmic}
\end{algorithm}
